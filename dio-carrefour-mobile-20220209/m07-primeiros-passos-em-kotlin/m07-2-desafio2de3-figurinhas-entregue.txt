// mÃ¡ximo divisor comum (chamada recursiva)
fun mdc(n1: Int, n2: Int): Int {
    var m : Int = n1
    var n : Int = n2
    var aux : Int

    while (n > 0) {
        aux = n
        n = m.rem(n)
        m = aux
    }
    return m
}
